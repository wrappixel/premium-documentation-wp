    Apex Charts

    ApexCharts is a modern JavaScript charting library designed to create interactive charts and visualizations using a simple and intuitive API. For comprehensive instructions and configuration options, please refer to the ApexCharts Documentation.

    Let me know if you want this tailored to a specific audience or tone (e.g., corporate, casual, developer-focused, etc.).

    Usage

    To use ApexCharts on your page, you must include the required vendor CSS files within the "Vendors CSS" section of the page header.

                  
  <link rel="stylesheet" href="@@webRoot/node_modules/apexcharts/dist/apexcharts.css">

    Include the following vendor scripts in the "Vendors JS" section of the page footer to enable ApexCharts functionality.

                  
  <script src="@@webRoot/node_modules/apexcharts/dist/apexcharts.min.js"></script>

var options_line = {
  series: [
    {
      name: "Desktops",
      data: [10, 41, 35, 51, 49, 62, 69, 91, 148],

    height: 350,
    type: "line",
    fontFamily: "inherit",
    zoom: {
      enabled: false,
    },
    toolbar: {
      show: false,
    },
  },
  dataLabels: {
    enabled: false,
  },
  colors: ["var(--bs-primary)"],
  stroke: {
    curve: "straight",
  },
  grid: {
    row: {
      colors: [ "transparent"], // takes an array which will be repeated on columns
      opacity: 0.5,
    },
    borderColor: "transparent",
  },
  xaxis: {
    categories: [
      "Jan",
      "Feb",
      "Mar",
      "Apr",
      "May",
      "Jun",
      "Jul",
      "Aug",
      "Sep",
    ],
    labels: {
      style: {
        colors: [
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
        ],
      },
    },
  },
  yaxis: {
    labels: {
      style: {
        colors: [
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
          "#a1aab2",
        ],
      },
    },
  },
  tooltip: {
    theme: "dark",
  },
};

var chart_line_basic = new ApexCharts(
  document.querySelector("#chart-line-basic"),
  options_line
);
chart_line_basic.render();
